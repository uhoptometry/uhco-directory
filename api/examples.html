How the Authorization header works

HTTP has a standard Authorization request header for passing credentials. The UHCO Directory API uses the Bearer scheme — you send your token as a plain string prefixed with the word Bearer and a space:

Authorization: Bearer uhcs_your_token_here

When protected data is also needed (Current-Student, Alumni, or Academic Programs orgs), add a second header:

Authorization: Bearer uhcs_your_token_here
X-API-Secret: uhcs_sec_your_secret_here
Headers are invisible to server logs and are never stored in browser history — always prefer them over putting credentials in the URL query string.

curl curl (command line)

The simplest way to test from a terminal. The -H flag sets a request header.

Basic request:

curl -H "Authorization: Bearer uhcs_your_token" \
     "http://127.0.0.1/api/v1/people?limit=5"

With a secret (to include protected data):

curl -H "Authorization: Bearer uhcs_your_token" \
     -H "X-API-Secret: uhcs_sec_your_secret" \
     "http://127.0.0.1/api/v1/people?flag=Current-Student&limit=10"

Pretty-print JSON output (requires python3):

curl -s -H "Authorization: Bearer uhcs_your_token" \
     "http://127.0.0.1/api/v1/people?limit=5" | python3 -m json.tool

JS JavaScript — browser fetch()

The browser's built-in fetch() function accepts a headers object. The API must be on the same origin (or allow CORS) for this to work from a browser page.

Basic fetch:

const token = 'uhcs_your_token';

const response = await fetch('/api/v1/people?limit=10', {
  headers: {
    'Authorization': `Bearer ${token}`
  }
});

if (!response.ok) {
  throw new Error(`API error: ${response.status}`);
}

const data = await response.json();
console.log(`Total records: ${data.TOTAL}`);
console.log(data.DATA);

With a secret for protected data:

const token  = 'uhcs_your_token';
const secret = 'uhcs_sec_your_secret';

const params = new URLSearchParams({
  flag:  'Current-Student',
  limit: 25
});

const response = await fetch(`/api/v1/people?${params}`, {
  headers: {
    'Authorization': `Bearer ${token}`,
    'X-API-Secret':  secret
  }
});

const data = await response.json();
console.log(data);

Reusable helper function:

const API_BASE   = 'http://127.0.0.1/api/v1';
const API_TOKEN  = 'uhcs_your_token';
const API_SECRET = 'uhcs_sec_your_secret'; // omit if not needed

async function apiGet(path, params = {}) {
  const url = new URL(API_BASE + path);
  Object.entries(params).forEach(([k, v]) => url.searchParams.set(k, v));

  const headers = { 'Authorization': `Bearer ${API_TOKEN}` };
  if (API_SECRET) headers['X-API-Secret'] = API_SECRET;

  const res = await fetch(url, { headers });
  if (!res.ok) {
    const err = await res.json().catch(() => ({ error: res.statusText }));
    throw new Error(err.ERROR ?? err.error ?? res.statusText);
  }
  return res.json();
}

// Usage:
const people = await apiGet('/people', { search: 'smith', limit: 20 });
const person  = await apiGet('/people/42');
const orgs    = await apiGet('/organizations');

Node Node.js (built-in fetch, v18+)

Node 18 and later include the same fetch() API as the browser — no extra packages needed.

// api_example.mjs  (run with: node api_example.mjs)

const BASE   = 'http://127.0.0.1/api/v1';
const TOKEN  = 'uhcs_your_token';
const SECRET = 'uhcs_sec_your_secret';  // leave blank string if not needed

async function apiGet(path, params = {}) {
  const url = new URL(BASE + path);
  for (const [k, v] of Object.entries(params)) url.searchParams.set(k, v);

  const headers = { 'Authorization': `Bearer ${TOKEN}` };
  if (SECRET) headers['X-API-Secret'] = SECRET;

  const res = await fetch(url, { headers });
  if (!res.ok) {
    const body = await res.text();
    throw new Error(`HTTP ${res.status}: ${body}`);
  }
  return res.json();
}

// --- Examples ---

// 1. All people (first 10)
const list = await apiGet('/people', { limit: 10 });
console.log(`Total people: ${list.TOTAL}`);
list.DATA.forEach(p => console.log(`  ${p.LASTNAME}, ${p.FIRSTNAME}`));

// 2. Filter by flag (requires secret if flag is protected)
const students = await apiGet('/people', { flag: 'Current-Student', limit: 5 });
console.log(`\nCurrent students (visible): ${students.TOTAL}`);

// 3. Single person by ID
const person = await apiGet('/people/42');
console.log('\nPerson:', person.USER);

Python Python (requests library)

Install with pip install requests if you don't have it already.

# api_example.py

import requests

BASE   = "http://127.0.0.1/api/v1"
TOKEN  = "uhcs_your_token"
SECRET = "uhcs_sec_your_secret"  # omit or set to "" if not needed

# Build a reusable session so headers are sent automatically on every request
session = requests.Session()
session.headers["Authorization"] = f"Bearer {TOKEN}"
if SECRET:
    session.headers["X-API-Secret"] = SECRET

def api_get(path, **params):
    r = session.get(BASE + path, params=params)
    r.raise_for_status()   # raises HTTPError on 4xx/5xx
    return r.json()

# --- Examples ---

# 1. First 10 people
data = api_get("/people", limit=10)
print(f"Total: {data['TOTAL']}")
for p in data["DATA"]:
    print(f"  {p['LASTNAME']}, {p['FIRSTNAME']}")

# 2. Search by name
results = api_get("/people", search="jones", limit=5)
print(f"\nSearch 'jones': {results['TOTAL']} results")

# 3. Filter by flag (will be empty without secret if flag is protected)
students = api_get("/people", flag="Current-Student", limit=25)
print(f"\nCurrent students (visible): {students['TOTAL']}")

# 4. All organizations
orgs = api_get("/organizations")
for o in orgs["DATA"]:
    print(f"  [{o['ORGID']}] {o['ORGNAME']}")

CFML ColdFusion (cfhttp)

<cfhttp> supports custom request headers via <cfhttpparam type="header">. The example below calls the API from another ColdFusion page on the same server.

Tag syntax:

<cfset apiBase  = "http://127.0.0.1/api/v1">
<cfset apiToken = "uhcs_your_token">
<cfset apiSecret = "uhcs_sec_your_secret"><!--- leave blank if not needed --->

<cfhttp method="GET"
        url="#apiBase#/people"
        result="httpResult"
        timeout="15">

    <!--- Authorization header — this is how Bearer tokens are sent --->
    <cfhttpparam type="header" name="Authorization" value="Bearer #apiToken#">

    <!--- Secret header — only needed for protected data --->
    <cfif len(apiSecret)>
        <cfhttpparam type="header" name="X-API-Secret" value="#apiSecret#">
    </cfif>

    <!--- Query string parameters --->
    <cfhttpparam type="url" name="limit"  value="10">
    <cfhttpparam type="url" name="search" value="smith">

</cfhttp>

<cfif httpResult.statusCode neq "200 OK">
    <cfthrow message="API error: #httpResult.statusCode#">
</cfif>

<cfset result = deserializeJSON(httpResult.fileContent)>
<cfdump var="#result#">

Script syntax (ColdFusion 2016+):

// api_example.cfm - cfscript version

variables.apiBase   = "http://127.0.0.1/api/v1";
variables.apiToken  = "uhcs_your_token";
variables.apiSecret = "uhcs_sec_your_secret";

// Helper: call any API endpoint
function apiGet(path, struct params = {}) {
    var httpService = new http(
        method  = "GET",
        url     = variables.apiBase & arguments.path,
        timeout = 15
    );

    // Authorization header — required on every call
    httpService.addParam(type="header", name="Authorization",
                         value="Bearer #variables.apiToken#");

    // Secret header — only add when we have one
    if (len(variables.apiSecret)) {
        httpService.addParam(type="header", name="X-API-Secret",
                             value=variables.apiSecret);
    }

    // Add any query string params
    for (var k in arguments.params) {
        httpService.addParam(type="url", name=k, value=arguments.params[k]);
    }

    var result = httpService.send().getPrefix();

    if (result.statusCode neq "200 OK") {
        throw(message="API error: #result.statusCode#", detail=result.fileContent);
    }

    return deserializeJSON(result.fileContent);
}

// --- Examples ---

// All people (first 10)
var list = apiGet("/people", { limit: 10 });
writeOutput("Total: #list.TOTAL#<br>");

// Filter by flag
var staff = apiGet("/people", { flag: "Staff", limit: 20 });
writeOutput("Staff count: #staff.TOTAL#<br>");

// Protected data — returns 0 without a valid secret
var students = apiGet("/people", { flag: "Current-Student" });
writeOutput("Students visible: #students.TOTAL#<br>");

// Single person
var person = apiGet("/people/42");
writeDump(var=person.USER, expand=false);
